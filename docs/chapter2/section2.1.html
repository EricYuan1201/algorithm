206.反转链表

需要达到熟读背诵的情况。


public static ListNode reverseList(ListNode head) {
    ListNode pre = head;
    ListNode cur = head.next;
    pre.next = null;  //注意点
    ListNode temp = null;
    while (cur != null) {
        temp = cur.next;
        cur.next = pre;
        pre = cur;
        cur = temp;
    }
    return pre;
}
